31/// An image constructed from a slice of raw pixel data.
32///
33/// The `ImageRaw` struct can be used to construct an image from a slice
34/// of raw image data. The storage format is determined by the [`PixelColor`]
35/// type `C` and the [`DataOrder`] `O`.
36///
37/// For color types with less than 8 bits per pixels the start of each row is
38/// aligned to the next whole byte.
39///
40/// Details about the conversion of raw data to color types are explained in the
41/// [`raw` module documentation].
42///
43/// To draw an `ImageRaw` object it needs to be wrapped in an [`Image`] object.
44///
45/// # Examples
46///
47/// ## Draw a 1BPP image
48///
49/// This example creates an image from 1 bit per pixel data.
50///
51/// ```
52/// use embedded_graphics::{
53///     image::{Image, ImageRaw},
54///     pixelcolor::BinaryColor,
55///     prelude::*,
56/// };
57/// # use embedded_graphics::mock_display::MockDisplay as Display;
58///
59/// /// 12 x 5 pixel image with 1 bit per pixel.
60/// /// The data for each row is 12 bits long and is padded with zeros on the
61/// /// end because each row needs to contain a whole number of bytes.
62/// #[rustfmt::skip]
63/// const DATA: &[u8] = &[
64///     0b11101111, 0b0101_0000,
65///     0b10001000, 0b0101_0000,
66///     0b11101011, 0b0101_0000,
67///     0b10001001, 0b0101_0000,
68///     0b11101111, 0b0101_0000,
69/// ];
70///
71/// // The image dimensions and the format of the stored raw data must be specified
72/// // when the `new` function is called. The data format can, for example, be specified
73/// // by using the turbofish syntax.
74/// let raw_image = ImageRaw::<BinaryColor>::new(DATA, Size::new(12, 5)).unwrap();
75///
76/// let image = Image::new(&raw_image, Point::zero());
77///
78/// let mut display = Display::default();
79///
80/// image.draw(&mut display)?;
81/// # Ok::<(), core::convert::Infallible>(())
82/// ```
83///
84/// ## Draw an image that uses multibyte pixel encoding
85///
86/// Colors with more than one byte per pixel need an additional type annotation for the byte order.
87/// For convenience, the [`ImageRawBE`] and [`ImageRawLE`] type aliases can be used to abbreviate
88/// the type.
89///
90/// ```
91/// use embedded_graphics::{
92///     image::{Image, ImageRaw, ImageRawBE, ImageRawLE},
93///     pixelcolor::{
94///         raw::{BigEndianLsb0, LittleEndianMsb0},
95///         Rgb565, Rgb888,
96///     },
97///     prelude::*,
98/// };
99/// # const DATA: &[u8] = &[0x55; 8 * 8 * 3];
100///
101/// // Rgb888 image with 24 bits per pixel and big endian byte order
102/// let image1 = ImageRawBE::<Rgb888>::new(DATA, Size::new(8, 8)).unwrap();
103/// // or:
104/// let image2 = ImageRaw::<Rgb888, BigEndianLsb0>::new(DATA, Size::new(8, 8)).unwrap();
105/// # assert_eq!(image1, image2);
106///
107/// // Rgb565 image with 16 bits per pixel and little endian byte order
108/// let image1 = ImageRawLE::<Rgb565>::new(DATA, Size::new(12, 8)).unwrap();
109/// // or:
110/// let image2 = ImageRaw::<Rgb565, LittleEndianMsb0>::new(DATA, Size::new(12, 8)).unwrap();
111/// # assert_eq!(image1, image2);
112/// ```
113///
114/// [`raw` module documentation]: crate::pixelcolor::raw
115/// [`Image`]: crate::image::Image
116/// [`Drawable`]: crate::drawable::Drawable
117/// [`PixelColor`]: crate::pixelcolor::PixelColor
118/// [`ByteOrder`]: crate::pixelcolor::raw::ByteOrder
119#[derive(Copy, Clone, Eq, PartialEq, Ord, PartialOrd, Hash, Debug)]
120#[cfg_attr(feature = "defmt", derive(::defmt::Format))]
121pub struct ImageRaw<'a, C, O = LittleEndianMsb0>
122where
123    C: PixelColor,
124    O: DataOrder,
125{
126    /// Image data, packed as dictated by raw data type `C::Raw`
127    data: &'a [u8],
128
129    /// Image size in pixels
130    size: Size,
131
132    pixel_type: PhantomData<C>,
133    data_order: PhantomData<O>,
134}
135
136impl<'a, C, O> ImageRaw<'a, C, O>
137where
138    C: PixelColor,
139    O: DataOrder,
140{
141    /// Creates a new image.
142    ///
143    /// Returns an error if the length of the data size doesn't match the
144    /// expected size based on the given image dimensions and the color depth.
145    pub const fn new(data: &'a [u8], size: Size) -> Result<Self, ImageRawError> {
146        let expected_size =
147            bytes_per_row(size.width, C::Raw::BITS_PER_PIXEL) * size.height as usize;
148
149        if data.len() != expected_size {
150            return Err(ImageRawError::InvalidDataSize {
151                expected_data_size: expected_size,
152            });
153        }
154
155        Ok(Self {
156            data,
157            size,
158            pixel_type: PhantomData,
159            data_order: PhantomData,
160        })
161    }
162
163    /// Creates a new image in a const context.
164    ///
165    /// While [`ImageRaw::new`] is also `const` it cannot easily be used to
166    /// initialize a constant, because `unwrap()` isn't available in a const
167    /// context in stable Rust. This method provides an alternative that panics
168    /// instead of returning an error.
169    ///
170    /// # Panics
171    ///
172    /// Panics if the given slice of data doesn't have the correct length to
173    /// match the image size and bit depth.
174    pub const fn new_const(data: &'a [u8], size: Size) -> Self {
175        match Self::new(data, size) {
176            Ok(image) => image,
177            Err(ImageRawError::InvalidDataSize { .. }) => panic!("Invalid data size"),
178        }
179    }
180
181    /// Returns the actual row width in pixels.
182    ///
183    /// For images with less than 8 bits per pixel each row is padded to contain an integer number
184    /// of bytes. This method returns the width of each row including the padding pixels.
185    const fn data_width(&self) -> u32 {
186        if C::Raw::BITS_PER_PIXEL < 8 {
187            let pixels_per_byte = 8 / C::Raw::BITS_PER_PIXEL as u32;
188
189            bytes_per_row(self.size.width, C::Raw::BITS_PER_PIXEL) as u32 * pixels_per_byte
190        } else {
191            self.size.width
192        }
193    }
194}
195
196/// Returns the length of each row in bytes.
197const fn bytes_per_row(width: u32, bits_per_pixel: usize) -> usize {
198    (width as usize * bits_per_pixel + 7) / 8
199}

201#[maybe_async::maybe_async(AFIT)]
202impl<'a, C, O> ImageDrawable for ImageRaw<'a, C, O>
203where
204    C: PixelColor,
205    O: DataOrder,
206    RawDataSlice<'a, C::Raw, O>: IntoIterator<Item = C::Raw>,
207{
208    type Color = C;
209
210    async fn draw<D>(&self, target: &mut D) -> Result<(), D::Error>
211    where
212        D: DrawTarget<Color = C>,
213    {
214        let row_skip = self.data_width() - self.size.width;
215
216        target
217            .fill_contiguous(
218                &self.bounding_box(),
219                ContiguousPixels::new(self, self.size, 0, row_skip as usize),
220            )
221            .await
222    }
223
224    async fn draw_sub_image<D>(&self, target: &mut D, area: &Rectangle) -> Result<(), D::Error>
225    where
226        D: DrawTarget<Color = Self::Color>,
227    {
228        // Don't draw anything if `area` is zero sized or partially outside the image.
229        if area.is_zero_sized()
230            || area.top_left.x < 0
231            || area.top_left.y < 0
232            || area.top_left.x as u32 + area.size.width > self.size.width
233            || area.top_left.y as u32 + area.size.height > self.size.height
234        {
235            return Ok(());
236        }
237
238        let data_width = self.data_width() as usize;
239
240        let initial_skip = area.top_left.y as usize * data_width + area.top_left.x as usize;
241        let row_skip = data_width - area.size.width as usize;
242
243        target
244            .fill_contiguous(
245                &Rectangle::new(Point::zero(), area.size),
246                ContiguousPixels::new(self, area.size, initial_skip, row_skip),
247            )
248            .await
249    }
250}

262impl<'a, C, O> GetPixel for ImageRaw<'a, C, O>
263where
264    C: PixelColor,
265    O: DataOrder,
266    RawDataSlice<'a, C::Raw, O>: IntoIterator<Item = C::Raw>,
267{
268    type Color = C;
269
270    fn pixel(&self, p: Point) -> Option<Self::Color> {
271        if p.x < 0 || p.y < 0 || p.x >= self.size.width as i32 || p.y >= self.size.height as i32 {
272            return None;
273        }
274
275        RawDataSlice::new(self.data)
276            .into_iter()
277            .nth(p.x as usize + p.y as usize * self.data_width() as usize)
278            .map(|r| r.into())
279    }
280}

282struct ContiguousPixels<'a, C, O>
283where
284    C: PixelColor,
285    O: DataOrder,
286    RawDataSlice<'a, C::Raw, O>: IntoIterator<Item = C::Raw>,
287{
288    iter: <RawDataSlice<'a, C::Raw, O> as IntoIterator>::IntoIter,
289
290    remaining_x: u32,
291    width: u32,
292
293    remaining_y: u32,
294    row_skip: usize,
295}
296
297impl<'a, C, O> ContiguousPixels<'a, C, O>
298where
299    C: PixelColor,
300    O: DataOrder,
301    RawDataSlice<'a, C::Raw, O>: IntoIterator<Item = C::Raw>,
302{
303    fn new(image: &ImageRaw<'a, C, O>, size: Size, initial_skip: usize, row_skip: usize) -> Self {
304        let mut iter = RawDataSlice::new(image.data).into_iter();
305
306        if initial_skip > 0 {
307            iter.nth(initial_skip - 1);
308        }
309
310        // Set `remaining_y` to `0` if `width == 0` to prevent integer underflow in `next`.
311        let remaining_y = if size.width > 0 { size.height } else { 0 };
312
313        Self {
314            iter,
315            remaining_x: size.width,
316            width: size.width,
317            remaining_y,
318            row_skip,
319        }
320    }
321}
322
323impl<'a, C, O> Iterator for ContiguousPixels<'a, C, O>
324where
325    C: PixelColor,
326    O: DataOrder,
327    RawDataSlice<'a, C::Raw, O>: IntoIterator<Item = C::Raw>,
328{
329    type Item = C;
330
331    fn next(&mut self) -> Option<Self::Item> {
332        if self.remaining_x > 0 {
333            self.remaining_x -= 1;
334
335            self.iter.next()
336        } else {
337            if self.remaining_y == 0 {
338                return None;
339            }
340
341            self.remaining_y -= 1;
342            self.remaining_x = self.width - 1;
343
344            self.iter.nth(self.row_skip)
345        }
346        .map(|c| c.into())
347    }
348}
