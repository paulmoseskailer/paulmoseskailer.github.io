35/// Shared Display.
36pub struct SharedDisplay<D: SharableBufferedDisplay> {
37    /// The actual display, locked with mutex
38    pub real_display: Mutex<CriticalSectionRawMutex, D>,
39    partition_areas: heapless::Vec<Rectangle, MAX_APPS_PER_SCREEN>,
40
41    spawner: &'static Spawner,
42}

44impl<B, D> SharedDisplay<D>
45where
46    D: SharableBufferedDisplay<BufferElement = B>,
47{
48    /// Creates a new Shared Display from a real display.
49    pub fn new(real_display: D, spawner: Spawner) -> Self {
50        let spawner_ref: &'static Spawner = SPAWNER.init(spawner);
51        SharedDisplay {
52            real_display: Mutex::new(real_display),
53            partition_areas: heapless::Vec::new(),
54            spawner: spawner_ref,
55        }
56    }
57
58    async fn new_partition(
59        &mut self,
60        area: Rectangle,
61    ) -> Result<DisplayPartition<D>, NewPartitionError> {
62        let real_display: &mut D = &mut *self.real_display.lock().await;
63
64        // check area inside display
65        let parent_area = real_display.bounding_box();
66        if !(parent_area.contains(area.top_left)
67            && parent_area.contains(area.bottom_right().unwrap_or(area.top_left)))
68        {
69            return Err(NewPartitionError::OutsideParent);
70        }
71
72        // check area not overlapping with existing partition_areas
73        for p in self.partition_areas.iter() {
74            if p.intersection(&area).size != Size::new(0, 0) {
75                return Err(NewPartitionError::Overlaps);
76            }
77        }
78
79        let index = self.partition_areas.len();
80        let result = DisplayPartition::new(
81            index.try_into().unwrap(),
82            real_display.get_buffer(),
83            parent_area.size,
84            area,
85            &FLUSH_REQUESTS,
86        );
87
88        if result.is_ok() {
89            self.partition_areas.push(area).unwrap();
90        }
91
92        result
93    }

95    /// Launches a new app in an area of the screen.
96    ///
97    /// Returns an error if the area is not available, overlaps with existing apps or the screen
98    /// border.
99    pub async fn launch_new_app<F>(
100        &mut self,
101        mut app_fn: F,
102        area: Rectangle,
103    ) -> Result<(), NewPartitionError>
104    where
105        F: AsyncFnMut(DisplayPartition<D>),
106        for<'b> F::CallRefFuture<'b>: 'static,
107    {
108        let partition = self.new_partition(area).await?;
109
110        let fut = app_fn(partition);
111        self.spawner.must_spawn(launch_future(Box::pin(fut), area));
112
113        Ok(())
114    }

116    /// Launches a new app that can launch other apps in an area of the screen.
117    ///
118    /// See [`launch_app_in_app`].
119    /// Returns an error if the area is not available, overlaps with existing apps or the screen
120    /// border.
121    pub async fn launch_new_recursive_app<F>(
122        &mut self,
123        mut app_fn: F,
124        area: Rectangle,
125    ) -> Result<(), NewPartitionError>
126    where
127        F: AsyncFnMut(DisplayPartition<D>, &'static Spawner) -> (),
128        for<'b> F::CallRefFuture<'b>: 'static,
129    {
130        let partition = self.new_partition(area).await?;
131
132        let fut = app_fn(partition, self.spawner);
133        self.spawner.must_spawn(launch_future(Box::pin(fut), area));
134
135        Ok(())
136    }

138    /// Runs a given flush function in a loop.
139    ///
140    /// Provides the passed in function with a Rectangle of the area that has been drawn to since
141    /// the last flush.
142    /// Only exits if the flush function returns [`FlushResult::Abort`].
143    pub async fn run_flush_loop_with<F>(&self, mut flush_area_fn: F, flush_interval: Duration)
144    where
145        F: AsyncFnMut(&mut D, Rectangle) -> FlushResult,
146    {
147        'flush: loop {
148            for partition in 0..self.partition_areas.len() {
149                let area_to_flush = self.partition_areas[partition];
150                let flush_result =
151                    flush_area_fn(&mut *self.real_display.lock().await, area_to_flush).await;
152                if flush_result == FlushResult::Abort {
153                    break 'flush;
154                }
155            }
156            Timer::after(flush_interval).await;
157        }
158    }
159
160    /// Spawns a background task that waits for flush requests from all [`DisplayPartition`]s and flushes.
161    pub async fn wait_for_flush_requests<F>(&self, mut flush_area_fn: F)
162    where
163        F: AsyncFnMut(&mut D, Rectangle) -> FlushResult,
164    {
165        'flush: loop {
166            let partition = FLUSH_REQUESTS.receive().await;
167            let area_to_flush = self.partition_areas[partition as usize];
168            let flush_result =
169                flush_area_fn(&mut *self.real_display.lock().await, area_to_flush).await;
170            if flush_result == FlushResult::Abort {
171                break 'flush;
172            }
173        }
174    }
175}

177#[embassy_executor::task(pool_size = MAX_APPS_PER_SCREEN)]
178pub(crate) async fn launch_future(app_future: Pin<Box<dyn Future<Output = ()>>>, area: Rectangle) {
179    app_future.await;
180
181    EVENTS.send(AppEvent::AppClosed(area)).await;
182}

184/// Launches an app from inside another app.
185pub async fn launch_app_in_app<F, D>(
186    spawner: &'static Spawner,
187    mut app_fn: F,
188    partition: DisplayPartition<D>,
189) where
190    D: SharableBufferedDisplay,
191    F: AsyncFnMut(DisplayPartition<D>) -> (),
192    for<'b> F::CallRefFuture<'b>: 'static,
193{
194    let area = partition.area;
195    let fut = app_fn(partition);
196    spawner.must_spawn(launch_future(Box::pin(fut), area));
